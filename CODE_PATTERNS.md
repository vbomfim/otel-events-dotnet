## 3. RUNTIME IMPLEMENTATION

### OtelEventsHealthCheckPublisher.cs

```csharp
using System.Collections.Concurrent;
using Microsoft.Extensions.Diagnostics.HealthChecks;
using Microsoft.Extensions.Logging;
using OtelEvents.HealthChecks.Events;

namespace OtelEvents.HealthChecks;

/// <summary>
/// IHealthCheckPublisher that emits schema-defined events for health check
/// execution results and state changes.
/// </summary>
internal sealed class OtelEventsHealthCheckPublisher : IHealthCheckPublisher
{
    private const int MaxPreviousStatesCapacity = 1000;

    private readonly ILogger<OtelEventsHealthCheckEventSource> _logger;
    private readonly OtelEventsHealthCheckOptions _options;
    private readonly ConcurrentDictionary<string, HealthStatus> _previousStates = new();

    public OtelEventsHealthCheckPublisher(
        ILogger<OtelEventsHealthCheckEventSource> logger,
        OtelEventsHealthCheckOptions options)
    {
        ArgumentNullException.ThrowIfNull(logger);
        ArgumentNullException.ThrowIfNull(options);

        _logger = logger;
        _options = options;
    }

    public Task PublishAsync(HealthReport report, CancellationToken cancellationToken)
    {
        ArgumentNullException.ThrowIfNull(report);

        foreach (var entry in report.Entries)
        {
            // Emit health.check.executed (every poll)
            if (_options.EmitExecutedEvents &&
                !(_options.SuppressHealthyExecutedEvents && entry.Value.Status == HealthStatus.Healthy))
            {
                _logger.EmitHealthCheckExecuted(
                    healthComponent: entry.Key,
                    healthStatus: entry.Value.Status.ToStringFast(),
                    healthDurationMs: entry.Value.Duration.TotalMilliseconds,
                    healthDescription: entry.Value.Description);
            }

            // Emit health.state.changed (only on transitions)
            if (_options.EmitStateChangedEvents)
            {
                var previousStatus = _previousStates.GetOrAdd(entry.Key, entry.Value.Status);
                if (previousStatus != entry.Value.Status)
                {
                    // State transition detected
                    if (_previousStates.Count >= MaxPreviousStatesCapacity &&
                        !_previousStates.ContainsKey(entry.Key))
                    {
                        _logger.LogWarning(
                            "Health check publisher state dictionary at capacity ({Capacity}). " +
                            "Check name '{CheckName}' not tracked. " +
                            "Health check component names should be static and finite.",
                            MaxPreviousStatesCapacity, entry.Key);
                    }
                    else
                    {
                        _logger.EmitHealthStateChanged(
                            healthComponent: entry.Key,
                            healthPreviousStatus: previousStatus.ToStringFast(),
                            healthStatus: entry.Value.Status.ToStringFast(),
                            healthDurationMs: entry.Value.Duration.TotalMilliseconds,
                            healthDescription: entry.Value.Description);

                        _previousStates[entry.Key] = entry.Value.Status;
                    }
                }
            }
        }

        // Emit health.report.completed (aggregate)
        if (_options.EmitReportCompletedEvents)
        {
            _logger.EmitHealthReportCompleted(
                healthOverallStatus: report.Status.ToStringFast(),
                healthTotalChecks: report.Entries.Count,
                healthDurationMs: report.TotalDuration.TotalMilliseconds);
        }

        return Task.CompletedTask;
    }
}
```

### OtelEventsHealthCheckOptions.cs

```csharp
namespace OtelEvents.HealthChecks;

/// <summary>
/// Configuration options for OtelEvents.HealthChecks integration pack.
/// </summary>
public class OtelEventsHealthCheckOptions
{
    /// <summary>
    /// Emit health.check.executed event for every health check poll cycle.
    /// Default: true
    /// </summary>
    public bool EmitExecutedEvents { get; set; } = true;

    /// <summary>
    /// Emit health.state.changed event only when a health check status transitions.
    /// Default: true
    /// </summary>
    public bool EmitStateChangedEvents { get; set; } = true;

    /// <summary>
    /// Emit health.report.completed event for the aggregate health report.
    /// Default: true
    /// </summary>
    public bool EmitReportCompletedEvents { get; set; } = true;

    /// <summary>
    /// When true, only emit health.check.executed for non-Healthy checks.
    /// Healthy checks still appear in health.state.changed on transitions.
    /// Default: false
    /// </summary>
    public bool SuppressHealthyExecutedEvents { get; set; } = false;

    /// <summary>
    /// Enable AllCausalityContext scope for events emitted by this publisher.
    /// Default: true
    /// </summary>
    public bool EnableCausalScope { get; set; } = true;
}
```

### OtelEventsHealthCheckExtensions.cs

```csharp
using Microsoft.Extensions.DependencyInjection;
using Microsoft.Extensions.Diagnostics.HealthChecks;
using Microsoft.Extensions.Logging;

namespace OtelEvents.HealthChecks;

/// <summary>
/// Extension methods for registering the OtelEvents.HealthChecks integration pack.
/// </summary>
public static class OtelEventsHealthCheckExtensions
{
    /// <summary>
    /// Registers the OtelEvents.HealthChecks integration pack.
    /// </summary>
    /// <param name="services">The service collection to configure.</param>
    /// <param name="configure">Optional action to configure options.</param>
    /// <returns>The service collection for chaining.</returns>
    public static IServiceCollection AddOtelEventsHealthChecks(
        this IServiceCollection services,
        Action<OtelEventsHealthCheckOptions>? configure = null)
    {
        ArgumentNullException.ThrowIfNull(services);

        var options = new OtelEventsHealthCheckOptions();
        configure?.Invoke(options);

        services.AddSingleton(options);
        services.AddSingleton<IHealthCheckPublisher>(
            sp => new OtelEventsHealthCheckPublisher(
                sp.GetRequiredService<ILogger<OtelEventsHealthCheckEventSource>>(),
                options));

        return services;
    }
}
```

### Unit Test: State Change Detection

```csharp
using Microsoft.Extensions.Diagnostics.HealthChecks;
using Microsoft.Extensions.Logging;
using OtelEvents.HealthChecks;
using OtelEvents.HealthChecks.Events;

namespace OtelEvents.HealthChecks.Tests;

public class OtelEventsHealthCheckPublisherTests : IDisposable
{
    private readonly TestLogExporter _exporter;
    private readonly ILoggerFactory _loggerFactory;
    private readonly OtelEventsHealthCheckPublisher _publisher;

    public OtelEventsHealthCheckPublisherTests()
    {
        _exporter = new TestLogExporter();
        _loggerFactory = LoggerFactory.Create(builder =>
        {
            builder.AddOpenTelemetry(options =>
            {
                options.AddProcessor(new SimpleLogRecordExportProcessor(_exporter));
            });
            builder.SetMinimumLevel(LogLevel.Trace);
        });

        var logger = _loggerFactory.CreateLogger<OtelEventsHealthCheckEventSource>();
        var options = new OtelEventsHealthCheckOptions();
        _publisher = new OtelEventsHealthCheckPublisher(logger, options);
    }

    public void Dispose()
    {
        _loggerFactory.Dispose();
    }

    [Fact]
    public async Task PublishAsync_FirstPoll_DoesNotEmitStateChanged()
    {
        // Arrange
        var report = new HealthReport(
            new Dictionary<string, HealthReportEntry>
            {
                ["Db"] = new HealthReportEntry(
                    status: HealthStatus.Healthy,
                    description: null,
                    duration: TimeSpan.FromMilliseconds(10),
                    exception: null,
                    data: new Dictionary<string, object>())
            },
            totalDuration: TimeSpan.FromMilliseconds(10));

        // Act
        await _publisher.PublishAsync(report, CancellationToken.None);

        // Assert
        var records = _exporter.GetRecords();
        var stateChangeEvent = records.FirstOrDefault(r => r.EventId.Id == 10402);
        Assert.Null(stateChangeEvent); // No state change on first poll
    }

    [Fact]
    public async Task PublishAsync_StatusTransition_EmitsStateChanged()
    {
        // Arrange
        var options = new OtelEventsHealthCheckOptions { EmitStateChangedEvents = true };
        var logger = _loggerFactory.CreateLogger<OtelEventsHealthCheckEventSource>();
        var publisher = new OtelEventsHealthCheckPublisher(logger, options);

        // First poll: Healthy
        var report1 = new HealthReport(
            new Dictionary<string, HealthReportEntry>
            {
                ["Db"] = new HealthReportEntry(
                    status: HealthStatus.Healthy,
                    description: null,
                    duration: TimeSpan.FromMilliseconds(10),
                    exception: null,
                    data: new Dictionary<string, object>())
            },
            totalDuration: TimeSpan.FromMilliseconds(10));

        // Second poll: Degraded
        var report2 = new HealthReport(
            new Dictionary<string, HealthReportEntry>
            {
                ["Db"] = new HealthReportEntry(
                    status: HealthStatus.Degraded,
                    description: "Slow response",
                    duration: TimeSpan.FromMilliseconds(5000),
                    exception: null,
                    data: new Dictionary<string, object>())
            },
            totalDuration: TimeSpan.FromMilliseconds(5000));

        // Act
        await publisher.PublishAsync(report1, CancellationToken.None);
        _exporter.Reset();
        await publisher.PublishAsync(report2, CancellationToken.None);

        // Assert
        var records = _exporter.GetRecords();
        var stateChangeEvent = records.FirstOrDefault(r => r.EventId.Id == 10402);
        Assert.NotNull(stateChangeEvent);
        Assert.Contains("health.state.changed", stateChangeEvent.Body?.ToString() ?? "");
    }

    [Fact]
    public async Task PublishAsync_SuppressHealthyExecutedEvents_DoesNotEmitHealthyChecks()
    {
        // Arrange
        var options = new OtelEventsHealthCheckOptions
        {
            EmitExecutedEvents = true,
            SuppressHealthyExecutedEvents = true
        };
        var logger = _loggerFactory.CreateLogger<OtelEventsHealthCheckEventSource>();
        var publisher = new OtelEventsHealthCheckPublisher(logger, options);

        var report = new HealthReport(
            new Dictionary<string, HealthReportEntry>
            {
                ["Healthy"] = new HealthReportEntry(
                    status: HealthStatus.Healthy,
                    description: null,
                    duration: TimeSpan.FromMilliseconds(10),
                    exception: null,
                    data: new Dictionary<string, object>()),
                ["Degraded"] = new HealthReportEntry(
                    status: HealthStatus.Degraded,
                    description: "Issue",
                    duration: TimeSpan.FromMilliseconds(50),
                    exception: null,
                    data: new Dictionary<string, object>())
            },
            totalDuration: TimeSpan.FromMilliseconds(60));

        // Act
        await publisher.PublishAsync(report, CancellationToken.None);

        // Assert
        var records = _exporter.GetRecords();
        var executedEvents = records.Where(r => r.EventId.Id == 10401).ToList();
        Assert.Single(executedEvents); // Only Degraded, not Healthy
    }
}
```

```csharp
        services.AddLogging(builder =>

            builder.SetMinimumLevel(LogLevel.Debug);

        });

        services.AddHealthChecks()
            .AddCheck("TestDb", () => HealthCheckResult.Healthy());

        services.AddOtelEventsHealthChecks(options =>
        {
            options.EmitExecutedEvents = true;
            options.EmitStateChangedEvents = true;
            options.EmitReportCompletedEvents = true;
        });

        _serviceProvider = services.BuildServiceProvider();
    }

    public void Dispose()

    }

    [Fact]
    public async Task FullPipeline_EmitsEventsWithCorrectShape()
    {
        // Arrange
        var healthCheckService = _serviceProvider.GetRequiredService<HealthCheckService>();
        var report = await healthCheckService.CheckHealthAsync();

        // Assert - Health check system will call IHealthCheckPublisher automatically
        var records = _exporter.GetRecords();
        
        var executedEvent = records.FirstOrDefault(r => r.EventId.Id == 10401);
        Assert.NotNull(executedEvent);
        
        var reportEvent = records.FirstOrDefault(r => r.EventId.Id == 10403);
        Assert.NotNull(reportEvent);
    }
}
```
